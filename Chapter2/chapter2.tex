\subsection{Poly VCR Profile Detection Algorithm}

\subsubsection{Integer Linear Programming}
The only input is \textit{ApprovalSet} of each voter. \\
Represented by $2D$ matrix $A_{vc}$

\subsubsection{Variables}

$x_c$ - candidate $c$ position on $x$ axis \\
$x_v$ - voter $v$ position on $x$ axis \\
$r_c$ - candidate $c$ radius \\
$r_v$ - voter $v$ radius 

\subsubsection{Transform VCR condition to Linear constraints}
\begin{equation}
    \begin{split}
    VCR & = \\
     & = \sqrt{(x_c - x_v)^2} \le r_c + r_v \\
     & = \|x_c - x_v\| \le r_c + r_v \\
     & = x_c - x_v \le r_c + r_v \ or -(x_c - x_v) \le r_c + r_v \\
     & = -(r_c + r_v) \le x_c - x_v \le r_c + r_v 
    \end{split}
\end{equation}

\subsubsection{Positive and Negative Condition and Big M method}

Introduce large constant positive parmater \\
$M$\\\\
Introduce new binary variables for each pair of candidate and voter:\\
$z1_{cv}\\ z2_{cv}$

\begin{align*}
\textit{Minimize 0} \\ 
\textit{foreach (c,v) in } C \times V : \\\textit{if } c \in ApprovalSet(v) \textit{ add constraints} \\ 
x_c - x_v \le r_c + r_v \\
-(x_c - x_v) \le r_c + r_v \\
\textit{if } c \not\in ApprovalSet(v) \textit{ add constraints} \\
x_c - x_v  + M(1 - z1_{cv}) \ge r_c + r_v + 1 \\
-(x_c - x_v) + M(1 - z2_{cv}) \ge r_c + r_v + 1 \\
z1_{cv} + z2_{cv} >= 1
\end{align*}

\subsubsection{Mock Distance Variable Condition}
Introduce new variable\\
$D_{cv}$ - distance between candidate $c$ and voter $v$ on $x$ axis

\begin{align*}
\textit{Minimize 0} \\ 
\textit{foreach (c,v) in } C \times V : \\
\textit{ add constraints} \\ 
D_{cv} = x_c - x_v \\
D_{cv} = -(x_c - x_v) \\
\textit{if } c \in ApprovalSet(v) \textit{ add constraints} \\ 
D_{cv} \le r_c + r_v \\
\textit{if } c \not\in ApprovalSet(v) \textit{ add constraints} \\
D_{cv} \ge r_c + r_v + 1 \\
\end{align*}
