\subsection{AV-CC-DV for Single Peaked ~(VR = CI = row C1P = $r_c=0$)}

Algorithm from \cite{control-sp-shield-faliszewski-hemaspaandra}

\begin{enumerate}
    \item kandydat $p$ na osi, pozostali na lewo i prawo
    \item wybór przeciwników (groźny kandydat): 
    \begin{enumerate}
        \item pierwszy na lewo $c_L$ i pierwszy na prawo $c_R$ takich, że $score(c_?) \geq score(p)$
        \item powtarzamy dla kolejnych lewych $c_{L_{i}}$ takich, że $score(c_{L_{i}}) > score(c_{L_{i-1}})$
        \item analogicznie dla prawych $c_{R_{i}}$ takich, że $score(c_{R_{i}}) > score(c_{R_{i-1}})$
    \end{enumerate}
    \item usuwanie głosów - zaczynając od najbliższego prawego przeciwnika $c_R$ (dla lewych - odbicie lustrzane)
    \begin{enumerate}
        \item notacja $v$: $b(v)$ początek przedziału (lewy) i $e(v)$ koniec przedziały (prawy)
        \item wybieramy głosujących $\widehat{V}$ takich że $\{v \in V | \: b(v) \in (p,c_R]\}$
        \item sortujemy $\widehat{V}$ malejąco po $e(v)$ (od najbardziej prawego końca przedziału)
        \item usuwamy $scoreDelta(p, c_R) + 1$ głosów
    \end{enumerate}
    \item powtarzamy do skutu, przeliczając aprobaty co iterację
\end{enumerate}


\subsection{AV-CC-DV for Candidate Range = VI = column C1P = $r_v=0$}

Algorithm from \cite{control-cr-how-hard-magiera-faliszewski}

\begin{enumerate}
    \item notacja CR dla $c$: $v_{f(c)}$ początek przedziału (lewy/górny) i $v_{e(c)}$ koniec przedziału (prawy/dolny)
    \item kandydat $p$
    \item Głosujący - wybieramy głosujących $V' = \{v \in V | \: p \notin A(v)\}$ 
    \item Przeciwnicy -  wybieramy kandydatów $\widehat{C} = \{c \in C \setminus \{p\} | \: scoreDelta(c,p) \geq 0\}$
    \item Usuwanie głosów 
    \begin{enumerate}
        \item zaczynając od przeciwnika $c_N$ takiego, że $e(c_N)$ jest minimalne
        \item usuwamy $deltaScore(c_N,p)+1$ głosów, zaczynając od końca przedziału $v_{e(c_{N})}$
    \end{enumerate} 
    \item powtarzamy do skutu, przeliczając $V'$ i $\widehat{C}$ co iterację
\end{enumerate}
